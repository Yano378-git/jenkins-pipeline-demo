pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo '🚧 [BUILD] Membangun project...'
                sh 'mkdir -p build'
                sh 'cp index.html build/'
            }
        }

        stage('Test') {
            steps {
                echo '🧪 [TEST] Mengecek hasil build...'
                sh 'ls build'
            }
        }

        stage('Deploy') {
            steps {
                echo '🚀 [DEPLOY] Menyalin hasil ke folder web server...'
                // Ganti folder di bawah sesuai lokasi web server kamu
                sh 'sudo mkdir -p /var/www/html/jenkins-demo'
                sh 'sudo cp -r build/* /var/www/html/jenkins-demo/'
            }
        }
    }

    post {
        success {
            echo '✅ Pipeline selesai dengan sukses!'
        }
        failure {
            echo '❌ Pipeline gagal dijalankan!'
        }
    }
}
